Basic Invoice Registration

Complete Node.js Example

const https = require('https');
const fs = require('fs');

class AEATClient {
  constructor(config) {
    this.config = {
      hostname: 'prewww1.aeat.es',
      port: 443,
      path: '/wlpl/TIKE-CONT/ws/SistemaFacturacion/VerifactuSOAP',
      ...config
    };
    
    this.certificate = {
      key: fs.readFileSync(config.keyPath),
      cert: fs.readFileSync(config.certPath),
      ca: fs.readFileSync(config.caPath)
    };
  }
  
  async registerInvoice(invoice) {
    try {
      const soapBody = this.createRegisterRequest(invoice);
      const response = await this.makeRequest(soapBody, 'urn:VeriFactu/Alta');
      return this.parseResponse(response);
    } catch (error) {
      throw new Error(`AEAT Registration failed: ${error.message}`);
    }
  }
  
  async cancelInvoice(invoice) {
    try {
      const soapBody = this.createCancelRequest(invoice);
      const response = await this.makeRequest(soapBody, 'urn:VeriFactu/Baja');
      return this.parseResponse(response);
    } catch (error) {
      throw new Error(`AEAT Cancellation failed: ${error.message}`);
    }
  }
  
  async queryInvoice(invoice) {
    try {
      const soapBody = this.createQueryRequest(invoice);
      const response = await this.makeRequest(soapBody, 'urn:VeriFactu/Consulta');
      return this.parseResponse(response);
    } catch (error) {
      throw new Error(`AEAT Query failed: ${error.message}`);
    }
  }
  
  createRegisterRequest(invoice) {
    return `<?xml version="1.0" encoding="UTF-8"?>
<soap:Envelope xmlns:soap="http://schemas.xmlsoap.org/soap/envelope/"
               xmlns:verifactu="urn:VeriFactu">
  <soap:Body>
    <verifactu:Alta>
      <verifactu:Cabecera>
        <verifactu:NIF>${invoice.nif}</verifactu:NIF>
        <verifactu:Fecha>${invoice.fecha}</verifactu:Fecha>
        <verifactu:Hora>${invoice.hora}</verifactu:Hora>
        <verifactu:Serie>${invoice.serie || ''}</verifactu:Serie>
        <verifactu:NumFactura>${invoice.numero}</verifactu:NumFactura>
      </verifactu:Cabecera>
      <verifactu:Factura>
        <verifactu:ImporteTotal>${invoice.importeTotal}</verifactu:ImporteTotal>
        <verifactu:ImporteTotalConImpuestos>${invoice.importeTotalConImpuestos}</verifactu:ImporteTotalConImpuestos>
        <verifactu:TipoFactura>${invoice.tipoFactura}</verifactu:TipoFactura>
        <verifactu:ClaveRegimenEspecialOTrascendencia>${invoice.claveRegimen}</verifactu:ClaveRegimenEspecialOTrascendencia>
      </verifactu:Factura>
    </verifactu:Alta>
  </soap:Body>
</soap:Envelope>`;
  }
  
  createCancelRequest(invoice) {
    return `<?xml version="1.0" encoding="UTF-8"?>
<soap:Envelope xmlns:soap="http://schemas.xmlsoap.org/soap/envelope/"
               xmlns:verifactu="urn:VeriFactu">
  <soap:Body>
    <verifactu:Baja>
      <verifactu:Cabecera>
        <verifactu:NIF>${invoice.nif}</verifactu:NIF>
        <verifactu:Fecha>${invoice.fecha}</verifactu:Fecha>
        <verifactu:Hora>${invoice.hora}</verifactu:Hora>
        <verifactu:Serie>${invoice.serie || ''}</verifactu:Serie>
        <verifactu:NumFactura>${invoice.numero}</verifactu:NumFactura>
      </verifactu:Cabecera>
      <verifactu:Factura>
        <verifactu:ImporteTotal>${invoice.importeTotal}</verifactu:ImporteTotal>
        <verifactu:ImporteTotalConImpuestos>${invoice.importeTotalConImpuestos}</verifactu:ImporteTotalConImpuestos>
        <verifactu:TipoFactura>${invoice.tipoFactura}</verifactu:TipoFactura>
        <verifactu:ClaveRegimenEspecialOTrascendencia>${invoice.claveRegimen}</verifactu:ClaveRegimenEspecialOTrascendencia>
      </verifactu:Factura>
    </verifactu:Baja>
  </soap:Body>
</soap:Envelope>`;
  }
  
  createQueryRequest(invoice) {
    return `<?xml version="1.0" encoding="UTF-8"?>
<soap:Envelope xmlns:soap="http://schemas.xmlsoap.org/soap/envelope/"
               xmlns:verifactu="urn:VeriFactu">
  <soap:Body>
    <verifactu:Consulta>
      <verifactu:Cabecera>
        <verifactu:NIF>${invoice.nif}</verifactu:NIF>
        <verifactu:Fecha>${invoice.fecha}</verifactu:Fecha>
        <verifactu:Hora>${invoice.hora}</verifactu:Hora>
        <verifactu:Serie>${invoice.serie || ''}</verifactu:Serie>
        <verifactu:NumFactura>${invoice.numero}</verifactu:NumFactura>
      </verifactu:Cabecera>
      <verifactu:Factura>
        <verifactu:ImporteTotal>${invoice.importeTotal}</verifactu:ImporteTotal>
        <verifactu:ImporteTotalConImpuestos>${invoice.importeTotalConImpuestos}</verifactu:ImporteTotalConImpuestos>
        <verifactu:TipoFactura>${invoice.tipoFactura}</verifactu:TipoFactura>
        <verifactu:ClaveRegimenEspecialOTrascendencia>${invoice.claveRegimen}</verifactu:ClaveRegimenEspecialOTrascendencia>
      </verifactu:Factura>
    </verifactu:Consulta>
  </soap:Body>
</soap:Envelope>`;
  }
  
  makeRequest(soapBody, soapAction) {
    return new Promise((resolve, reject) => {
      const options = {
        ...this.config,
        ...this.certificate,
        headers: {
          'Content-Type': 'text/xml; charset=utf-8',
          'SOAPAction': soapAction
        }
      };
      
      const req = https.request(options, (res) => {
        let data = '';
        res.on('data', (chunk) => {
          data += chunk;
        });
        res.on('end', () => {
          resolve(data);
        });
      });
      
      req.on('error', (error) => {
        reject(error);
      });
      
      req.write(soapBody);
      req.end();
    });
  }
  
  parseResponse(xmlResponse) {
    const csvMatch = xmlResponse.match(/(.*?)<\/verifactu:CSV>/);
    const codigoMatch = xmlResponse.match(/(.*?)<\/verifactu:Codigo>/);
    const descripcionMatch = xmlResponse.match(/(.*?)<\/verifactu:Descripcion>/);
    
    return {
      csv: csvMatch ? csvMatch[1] : null,
      codigo: codigoMatch ? codigoMatch[1] : null,
      descripcion: descripcionMatch ? descripcionMatch[1] : null,
      success: codigoMatch && codigoMatch[1] === '0'
    };
  }
}

Database Integration

MySQL Integration Example

const mysql = require('mysql2/promise');

class InvoiceService {
  constructor(dbConfig) {
    this.db = mysql.createConnection(dbConfig);
  }
  
  async processInvoice(invoiceId) {
    try {
      // Get invoice from database
      const invoice = await this.getInvoice(invoiceId);
      
      // Register with AEAT
      const aeatResult = await this.registerWithAEAT(invoice);
      
      // Update database with result
      await this.updateInvoiceStatus(invoiceId, aeatResult);
      
      return aeatResult;
    } catch (error) {
      // Log error and update database
      await this.logError(invoiceId, error.message);
      throw error;
    }
  }
  
  async getInvoice(invoiceId) {
    const [rows] = await this.db.execute(
      'SELECT * FROM invoices WHERE id = ?',
      [invoiceId]
    );
    
    if (rows.length === 0) {
      throw new Error('Invoice not found');
    }
    
    return rows[0];
  }
  
  async updateInvoiceStatus(invoiceId, result) {
    const query = `
      UPDATE invoices 
      SET aeat_csv_code = ?, 
          aeat_status = ?, 
          aeat_error = NULL,
          updated_at = NOW()
      WHERE id = ?
    `;
    
    const status = result.success ? 'verified' : 'error';
    await this.db.execute(query, [result.csv, status, invoiceId]);
  }
  
  async logError(invoiceId, errorMessage) {
    const query = `
      UPDATE invoices 
      SET aeat_status = 'error', 
          aeat_error = ?,
          updated_at = NOW()
      WHERE id = ?
    `;
    
    await this.db.execute(query, [errorMessage, invoiceId]);
  }
}

Error Handling Examples

Comprehensive Error Handling

class AEATService {
  constructor(config) {
    this.config = config;
    this.retryAttempts = 3;
    this.retryDelay = 1000;
  }
  
  async registerInvoiceWithRetry(invoice) {
    let lastError;
    
    for (let attempt = 1; attempt <= this.retryAttempts; attempt++) {
      try {
        return await this.registerInvoice(invoice);
      } catch (error) {
        lastError = error;
        
        if (this.isRetryableError(error)) {
          console.log(`Attempt ${attempt} failed, retrying in ${this.retryDelay}ms...`);
          await this.delay(this.retryDelay * attempt);
          continue;
        } else {
          throw error;
        }
      }
    }
    
    throw new Error(`All ${this.retryAttempts} attempts failed. Last error: ${lastError.message}`);
  }
  
  isRetryableError(error) {
    const retryableErrors = [
      'ECONNRESET',
      'ETIMEDOUT',
      'ENOTFOUND',
      'ECONNREFUSED'
    ];
    
    return retryableErrors.some(code => error.code === code);
  }
  
  delay(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
